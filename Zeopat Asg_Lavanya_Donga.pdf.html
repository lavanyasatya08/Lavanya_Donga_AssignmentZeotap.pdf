import pandas as pd
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity
from sklearn.preprocessing import StandardScaler
from sklearn.decomposition import PCA
from sklearn.metrics import mean_squared_error
from math import sqrt

# Load the dataset
customers_df = pd.read_csv('Customers.csv')
products_df = pd.read_csv('Products.csv')
transactions_df = pd.read_csv('Transactions.csv')

# Merge datasets for customer-product interaction
merged_df = pd.merge(transactions_df, customers_df, on='CustomerID', how='left')
merged_df = pd.merge(merged_df, products_df, on='ProductID', how='left')

# Create a customer-product interaction matrix (CustomerID x ProductID)
interaction_matrix = pd.pivot_table(merged_df, values='Quantity', index='CustomerID', columns='ProductID', aggfunc='sum', fill_value=0)

# Normalize the interaction matrix using StandardScaler
scaler = StandardScaler()
interaction_matrix_scaled = scaler.fit_transform(interaction_matrix)

# Apply PCA for dimensionality reduction (optional for large datasets)
pca = PCA(n_components=20)
interaction_matrix_pca = pca.fit_transform(interaction_matrix_scaled)

# Calculate cosine similarity between customers
cos_sim = cosine_similarity(interaction_matrix_pca)

# Function to get the top 3 lookalike customers
def get_lookalikes(customer_id, top_n=3):
    customer_idx = interaction_matrix.index.get_loc(customer_id)
    similarity_scores = cos_sim[customer_idx]
    similar_customers = list(zip(interaction_matrix.index, similarity_scores))
    similar_customers.sort(key=lambda x: x[1], reverse=True)
    return similar_customers[1:top_n+1]  # Exclude the customer itself

# Create the Lookalike dictionary for customers C0001 to C0020
lookalike_data = {}
for customer_id in interaction_matrix.index[:20]:
    lookalike_data[customer_id] = get_lookalikes(customer_id)

# Convert the Lookalike data into a DataFrame
lookalike_records = []
for cust_id, similar_customers in lookalike_data.items():
    for similar_customer, score in similar_customers:
        lookalike_records.append([cust_id, similar_customer, score])

# Create a DataFrame with appropriate column names
lookalike_df = pd.DataFrame(lookalike_records, columns=['CustomerID', 'LookalikeCustomerID', 'SimilarityScore'])

# Save the result to a CSV file
lookalike_df.to_csv('Lookalike.csv', index=False)

# Optionally print the first few rows of the Lookalike DataFrame
print(lookalike_df.head())

simulated_ground_truth = {

# Evaluation Metrics - RMSE and MSE

def rmse(predicted, ground_truth):
    """Calculate RMSE between predicted and ground truth values."""
    # RMSE is the square root of MSE, so we calculate MSE first
    mse = mean_squared_error(predicted, ground_truth)
    return sqrt(mse)

def mse(predicted, ground_truth):
    """Calculate MSE between predicted and ground truth values."""
    return mean_squared_error(predicted, ground_truth)

import pandas as pd
import numpy as np
from sklearn.preprocessing import StandardScaler
from sklearn.metrics.pairwise import cosine_similarity

# Load datasets (replace with actual file paths)
customers_df = pd.read_csv('Customers.csv')  # Columns: CustomerID, Region
transactions_df = pd.read_csv('Transactions.csv')  # Columns: CustomerID, ProductID, Quantity

# Step 1: Create Customer Profile Features
region_features = pd.get_dummies(customers_df[['CustomerID', 'Region']], columns=['Region'])
region_features.set_index('CustomerID', inplace=True)

# Step 2: Create Transaction Features with TF-IDF
transaction_matrix = transactions_df.groupby(['CustomerID', 'ProductID'])['Quantity'].sum().unstack(fill_value=0)
transaction_totals = transaction_matrix.sum(axis=0)
customer_totals = transaction_matrix.sum(axis=1)
tf = transaction_matrix.div(customer_totals, axis=0)
idf = np.log(len(transaction_matrix) / (transaction_totals + 1))
tfidf_matrix = tf.multiply(idf, axis=1)

# Step 3: Combine Profile and Transaction Features
merged_features = region_features.join(tfidf_matrix, how='left').fillna(0)

# Step 4: Feature Scaling
scaler = StandardScaler()
scaled_features = scaler.fit_transform(merged_features)

# Step 5: Compute Cosine Similarity
cosine_sim = cosine_similarity(scaled_features)

# Step 6: Generate Recommendations
def get_top_k_similar(customers, similarity_matrix, top_k=3):
    recommendations = {}
    for idx, customer_id in enumerate(customers):
        similarities = list(enumerate(similarity_matrix[idx]))
        sorted_similarities = sorted(similarities, key=lambda x: x[1], reverse=True)[1:top_k+1]
        recommendations[customer_id] = [(customers[sim[0]], sim[1]) for sim in sorted_similarities]
    return recommendations

customer_ids = merged_features.index
recommendations = get_top_k_similar(customer_ids, cosine_sim, top_k=3)

# Create a Recommendations DataFrame
lookalike_data = {
    cust_id: [f"{sim_cust[0]}:{sim_cust[1]:.2f}" for sim_cust in sims]
    for cust_id, sims in recommendations.items()
}
lookalike_df = pd.DataFrame.from_dict(lookalike_data, orient='index', columns=['Top_1', 'Top_2', 'Top_3'])
lookalike_df.index.name = 'CustomerID'

def calculate_precision_recall(recommendations, ground_truth):
    precision_scores, recall_scores = [], []

    for cust_id, recs in recommendations.items():
        recommended_customers = {rec[0] for rec in recs}
        relevant_customers = ground_truth.get(cust_id, set())

        if not relevant_customers:
            continue

        true_positives = recommended_customers & relevant_customers
        precision = len(true_positives) / len(recommended_customers) if recommended_customers else 0
        recall = len(true_positives) / len(relevant_customers) if relevant_customers else 0

        precision_scores.append(precision)
        recall_scores.append(recall)

    avg_precision = sum(precision_scores) / len(precision_scores) if precision_scores else 0
    avg_recall = sum(recall_scores) / len(recall_scores) if recall_scores else 0
    return avg_precision, avg_recall
